  --config streaming_couping/configs/default.yaml \
  --manifest data/processed/scannetpp_pinhole_2d/manifest_rgb_cache.json \
  --scene-id 00a231a370 \
  --instance-id 37 \
  --frame-indices 133 162 520 566 477 \
  --sam3-device cuda:3 \

  --output-dir outputs/streaming_couping_paired_full_mask_memory_inst37

# Oracle feasibility test for the reverse SAM3 -> geometry direction.
# It compares the point head, translation-refined point head, and a camera-consistent
# depth-head unprojection. Each native geometry branch fixes its gauge once at the
# same reference frame; later-frame GT never enters alignment or object ICP.
PYTHONPATH=src:. python -m streaming_couping.scripts.run_gt_mask_pose_refinement \
  --config streaming_couping/configs/default.yaml \
  --manifest data/processed/scannetpp_pinhole_2d/manifest_rgb_cache.json \
  --scene-id 00a231a370 \
  --instance-id 37 \
  --frame-indices 133 162 520 566 477 \
  --reference-sequence-index 0 \
  --pose-refinement-mode translation_only \
  --geometry-device cuda:1 \
  --confidence-threshold 0.30 \
  --icp-max-correspondence 0.20 \
  --output-dir outputs/streaming_couping_gt_mask_depth_camera_diag_inst37
