# One-time repair for the five selected RGB frames. The summary fallback is
# debug-only when the original NAS files are not readable.
PYTHONPATH=src python scripts/cache_scannetpp_rgb.py \
  --manifest data/processed/scannetpp_pinhole_2d/manifest.json \
  --scene-ids 00a231a370 \
  --frame-indices 133 162 520 566 477 \
  --allow-summary-fallback \
  --output-manifest data/processed/scannetpp_pinhole_2d/manifest_rgb_cache.json

for prompt_mode in point box_point; do
  PYTHONPATH=src:. python -m streaming_couping.scripts.run_bridge \
    --config streaming_couping/configs/default.yaml \
    --manifest data/processed/scannetpp_pinhole_2d/manifest_rgb_cache.json \
    --scene-id 00a231a370 \
    --instance-id 37 \
    --frame-indices 133 162 520 566 477 \
    --sam3-device cuda:3 \
    --geometry-device cuda:1 \
    --geometry-modes zero aligned shuffled \
    --fallback-prompt-mode "${prompt_mode}" \
    --output-dir "outputs/streaming_couping_${prompt_mode}_controls_inst37"
done

# B0/B1/B2 memory ablation. Geometry is used once at the first accepted
# recovery; the complete B1 mask is written into SAM3 so B1/B2 are identical
# on the recovery frame. B2 then continues from SAM3 memory with the same ID.
PYTHONPATH=src:. python -m streaming_couping.scripts.run_bridge \
  --config streaming_couping/configs/default.yaml \
  --manifest data/processed/scannetpp_pinhole_2d/manifest_rgb_cache.json \
  --scene-id 00a231a370 \
  --instance-id 37 \
  --frame-indices 133 162 520 566 477 \
  --sam3-device cuda:3 \
  --geometry-device cuda:1 \
  --geometry-modes aligned \
  --fallback-prompt-mode box_point \
  --memory-writeback \
  --memory-writeback-prompt-mode mask \
  --output-dir outputs/streaming_couping_memory_full_mask_inst37
