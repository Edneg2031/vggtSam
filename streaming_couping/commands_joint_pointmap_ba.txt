validate_clip_export() {
  local clip="$1"
  local frames="$2"
  local root="$FINAL_EXPORT/$clip"
  local expected_mask_png=$((5 * frames + 1))
  local mask_kind
  local actual

  for mask_kind in \
    segmentation_masks \
    geometry_trusted_masks \
    raw_tracking_masks
  do
    actual=$(
      find "$root/$mask_kind" -type f -name '*.png' |
        wc -l |
        tr -d '[:space:]'
    )
    if [[ "$actual" != "$expected_mask_png" ]]; then
      echo "incomplete $mask_kind for $clip: $actual PNG, expected $expected_mask_png" >&2
      return 1
    fi
  done

  actual=$(
    find "$root" -type f -name '*.ply' |
      wc -l |
      tr -d '[:space:]'
  )
  if [[ "$actual" != "45" ]]; then
    echo "incomplete point-cloud export for $clip: $actual PLY, expected 45" >&2
    return 1
  fi

  for required in \
    "$root/comparison_gt_world/pose_comparison.png" \
    "$root/comparison_gt_world/pose_comparison.pdf" \
    "$root/comparison_gt_world/camera_poses.csv" \
    "$root/deployable_native/camera_poses.csv"
  do
    if [[ ! -f "$required" ]]; then
      echo "missing required export: $required" >&2
      return 1
    fi
  done
  echo "validated complete export: $clip"
}

validate_clip_export 00a231a370_90_240_37_68_54 7
